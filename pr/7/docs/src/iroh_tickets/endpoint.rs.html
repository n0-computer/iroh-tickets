10/// A token containing information for establishing a connection to an endpoint.
11///
12/// Contains
13/// - The [`EndpointId`] of the endpoint to connect to (a 32-byte ed25519 public key).
14/// - Any known [`TransportAddr`]s on which the endpoint can be reached.
15///
16/// This allows establishing a connection to the endpoint in most circumstances where it is
17/// possible to do so.
18///
19/// This [`EndpointTicket`] is a single item which can be easily serialized and deserialized and
20/// implements the [`Ticket`] trait.  The [`Display`] and [`FromStr`] traits can also be
21/// used to round-trip the ticket to string.
22///
23/// [`EndpointId`]: iroh_base::EndpointId
24/// [`Display`]: std::fmt::Display
25/// [`FromStr`]: std::str::FromStr
26/// [`TransportAddr`]: iroh_base::TransportAddr
27#[derive(Debug, Clone, PartialEq, Eq, derive_more::Display)]
28#[display("{}", Ticket::serialize(self))]
29pub struct EndpointTicket {
30    addr: EndpointAddr,
31}
32
33/// Wire format for [`EndpointTicket`].
34#[derive(Serialize, Deserialize)]
35enum TicketWireFormat {
36    Variant1(Variant1EndpointTicket),
37}
38
39#[derive(Serialize, Deserialize)]
40struct Variant1EndpointTicket {
41    addr: Variant1EndpointAddr,
42}
43
44impl Ticket for EndpointTicket {
45    const KIND: &'static str = "endpoint";
46
47    fn to_bytes(&self) -> Vec<u8> {
48        let data = TicketWireFormat::Variant1(Variant1EndpointTicket {
49            addr: Variant1EndpointAddr {
50                id: self.addr.id,
51                info: Variant1AddrInfo {
52                    addrs: self.addr.addrs.clone(),
53                },
54            },
55        });
56        postcard::to_stdvec(&data).expect("postcard serialization failed")
57    }
58
59    fn from_bytes(bytes: &[u8]) -> Result<Self, ParseError> {
60        let res: TicketWireFormat = postcard::from_bytes(bytes)?;
61        let TicketWireFormat::Variant1(Variant1EndpointTicket { addr }) = res;
62        Ok(Self {
63            addr: EndpointAddr {
64                id: addr.id,
65                addrs: addr.info.addrs,
66            },
67        })
68    }
69}
70
71impl FromStr for EndpointTicket {
72    type Err = ParseError;
73
74    fn from_str(s: &str) -> Result<Self, Self::Err> {
75        Ticket::deserialize(s)
76    }
77}

79impl EndpointTicket {
80    /// Creates a new ticket.
81    pub fn new(addr: EndpointAddr) -> Self {
82        Self { addr }
83    }
84
85    /// The [`EndpointAddr`] of the provider for this ticket.
86    pub fn endpoint_addr(&self) -> &EndpointAddr {
87        &self.addr
88    }
89}

91impl From<EndpointAddr> for EndpointTicket {
92    /// Creates a ticket from given addressing info.
93    fn from(addr: EndpointAddr) -> Self {
94        Self { addr }
95    }
96}
97
98impl From<EndpointTicket> for EndpointAddr {
99    /// Returns the addressing info from given ticket.
100    fn from(ticket: EndpointTicket) -> Self {
101        ticket.addr
102    }
103}

105impl Serialize for EndpointTicket {
106    fn serialize<S: serde::Serializer>(&self, serializer: S) -> Result<S::Ok, S::Error> {
107        if serializer.is_human_readable() {
108            serializer.serialize_str(&self.to_string())
109        } else {
110            let EndpointTicket { addr } = self;
111            (addr).serialize(serializer)
112        }
113    }
114}
115
116impl<'de> Deserialize<'de> for EndpointTicket {
117    fn deserialize<D: serde::Deserializer<'de>>(deserializer: D) -> Result<Self, D::Error> {
118        if deserializer.is_human_readable() {
119            let s = String::deserialize(deserializer)?;
120            Self::from_str(&s).map_err(serde::de::Error::custom)
121        } else {
122            let peer = Deserialize::deserialize(deserializer)?;
123            Ok(Self::new(peer))
124        }
125    }
126}

139#[cfg(test)]
140mod tests {
141    use std::net::{Ipv4Addr, SocketAddr};
142
143    use data_encoding::HEXLOWER;
144    use iroh_base::{PublicKey, SecretKey, TransportAddr};
145    use rand::SeedableRng;
146
147    use super::*;
148
149    fn make_ticket() -> EndpointTicket {
150        let mut rng = rand_chacha::ChaCha8Rng::seed_from_u64(0u64);
151        let peer = SecretKey::generate(&mut rng).public();
152        let addr = SocketAddr::from((Ipv4Addr::LOCALHOST, 1234));
153        EndpointTicket {
154            addr: EndpointAddr::from_parts(peer, [TransportAddr::Ip(addr)]),
155        }
156    }
157
158    #[test]
159    fn test_ticket_postcard() {
160        let ticket = make_ticket();
161        let bytes = postcard::to_stdvec(&ticket).unwrap();
162        let ticket2: EndpointTicket = postcard::from_bytes(&bytes).unwrap();
163        assert_eq!(ticket2, ticket);
164    }
165
166    #[test]
167    fn test_ticket_json() {
168        let ticket = make_ticket();
169        let json = serde_json::to_string(&ticket).unwrap();
170        let ticket2: EndpointTicket = serde_json::from_str(&json).unwrap();
171        assert_eq!(ticket2, ticket);
172    }
173
174    #[test]
175    fn test_ticket_base32() {
176        let endpoint_id =
177            PublicKey::from_str("ae58ff8833241ac82d6ff7611046ed67b5072d142c588d0063e942d9a75502b6")
178                .unwrap();
179
180        let ticket = EndpointTicket {
181            addr: EndpointAddr::from_parts(
182                endpoint_id,
183                [
184                    TransportAddr::Relay("http://derp.me./".parse().unwrap()),
185                    TransportAddr::Ip("127.0.0.1:1024".parse().unwrap()),
186                ],
187            ),
188        };
189        let base32 = data_encoding::BASE32_NOPAD
190            .decode(
191                ticket
192                    .to_string()
193                    .strip_prefix("endpoint")
194                    .unwrap()
195                    .to_ascii_uppercase()
196                    .as_bytes(),
197            )
198            .unwrap();
199        let expected = [
200            // variant
201            "00",
202            // endpoint id, 32 bytes, see above
203            "ae58ff8833241ac82d6ff7611046ed67b5072d142c588d0063e942d9a75502b6",
204            // two addrs
205            "02",
206            // TransportAddr: Relay
207            "00",
208            // 16 bytes
209            "10",
210            // RelayUrl
211            "687474703a2f2f646572702e6d652e2f",
212            // TransportAddr: IP
213            "01",
214            // IPv4
215            "00",
216            // address, see above
217            "7f0000018008",
218        ];
219
220        // 00ae58ff8833241ac82d6ff7611046ed67b5072d142c588d0063e942d9a75502b6
221        // 02
222        // 00
223        // 10
224        // 687474703a2f2f646572702e6d652e2f
225        // 01
226        // 00
227        // 7f0000018008
228        let expected = HEXLOWER.decode(expected.concat().as_bytes()).unwrap();
229        assert_eq!(base32, expected);
230    }
